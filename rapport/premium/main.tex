\section{Introduction}

\paragraph{Conformité au canevas.}
Ce document suit le plan imposé : sections \emph{Introduction} à \emph{Bilan}, puis bibliographie, webographie et annexes. Le format est A4 en 12~pt avec les marges du canevas. Les figures et tableaux sont cités explicitement dans le texte (par ex.\ figure~\ref{fig:coeur-dense} et tableau~\ref{tab:libs}).

\subsection{Contexte et positionnement}
Les politiques de financement de l'Union européenne produisent des \textbf{traces relationnelles riches}~: co-participation à des projets, montants engagés, paysage institutionnel et, via des référentiels complémentaires, des liens vers littérature scientifique et concepts thématiques. La modélisation en graphe permet d'extraire des \techterm{signaux structurels} au-delà de tableaux agrégés.

\begin{defbox}[Réseau de co-financement]
Un \textbf{graphe de collaboration explicite} $G_{\mathrm{ex}}=(V,E)$~: sommets $V$ = organisations, arête $\{u,v\}\in E$ s'il existe au moins un projet où $u$ et $v$ reçoivent une contribution européenne. Poids entier~: nombre de projets en commun.
\end{defbox}

\subsection{Vue d'ensemble du travail réalisé}
\begin{insightbox}
Chaîne complète~: \textbf{ingestion open data} (CORDIS, OpenAlex), \textbf{normalisation relationnelle}, construction d'un \textbf{graphe multiplexe} (collaboration explicite, similarité thématique, bipartitions financement/projet), calcul de \textbf{métriques de réseau} (PageRank, Louvain, betweenness, contrainte de Burt), et \textbf{visualisations interactives} (Folium, fiches HTML, carte pays, démo bipartite, suite multi-vues \texttt{interactive\_suite.html}).
\end{insightbox}

\subsection{Problématiques opérationnelles}
\begin{itemize}[leftmargin=1.2em]
  \item \textbf{Résilience d'identité :} déduplication robuste d'organisations (noms variables, identifiants hétérogènes).
  \item \textbf{Échelle :} millions d'évènements de co-présence potentiels—choix algorithmiques (approximation, filtrage d'affichage).
  \item \textbf{Hétérogénéité :} articuler \textit{fait financier} (CORDIS) et \textit{signal thématique} partiel ou externe (OpenAlex).
  \item \textbf{Opportunités de collaboration :} définir et prioriser les paires ``proches'' sur le graphe \emph{thématique} mais absentes du graphe \emph{explicite} CORDIS—sans confondre corrélation thématique et intention ou faisabilité contractuelle réelle ; le détail du calcul figure en section~\ref{sec:opportunites}.
  \item \textbf{Communication :} rendre les résultats navigables pour décideurs et analystes (\emph{storytelling} cartographique, suite HTML à onglets).
\end{itemize}

\section{Description du projet et objectifs}

\subsection{Périmètre et livrables}
Réaliser une chaîne \textbf{open data $\to$ tables normalisées $\to$ graphes multiplexes $\to$ métriques $\to$ visualisations} sur le financement européen~: identifier des organisations, leurs collaborations contractuelles (CORDIS), enrichir par publications et concepts (OpenAlex), calculer des centralités et communautés, produire une carte interactive (Folium), des fiches organisations, et une \textbf{interface unique} regroupant plusieurs HTML (\texttt{data/graphs/interactive\_suite.html}).

\subsection{Encadrement méthodologique}
La littérature des \textbf{réseaux scientifiques et d'innovation} mobilise souvent des bipartis acteurs--artefacts, des projections pondérées, et des indicateurs de centralité / modularité\footnote{Communautés (Louvain, Leiden), revues comme \cite{fortunato2010,newman2010}.}. Les données ouvertes imposent nettoyage d'identifiants et contrôle de qualité.

\subsection{Positionnement du multiplexe}
L'approche combine (i)~une couche \textbf{explicite} (co-financement)~; (ii)~une couche \textbf{latente} (profils thématiques / similarité)~; (iii)~des \textbf{scores de centralité} pour prioriser l'exploration.

\begin{insightbox}
Le multiplexe autorise des \textbf{recouvrements informatifs} (qui collabore \emph{vs} qui ressemble) et des \emph{écarts} exploitables—par ex.\ paires thématiquement proches sans arête CORDIS, pour recommandations d'opportunité.
\end{insightbox}

\section{Travail réalisé}

\subsection{Sources ouvertes : jeux téléchargés et fichiers bruts importés}

\subsubsection{CORDIS (Commission européenne)}
Le script \path{src/fetch_cordis.py} télécharge automatiquement les exports officiels (fichiers \texttt{.csv} ou archives \texttt{.zip}) depuis \url{https://cordis.europa.eu/data}~: jeux projets et organisations pour \textbf{Horizon~2020} (\texttt{cordis-h2020*\ldots}) et \textbf{Horizon Europe} (\texttt{cordis-he*\ldots}, variantes de casse dans les URL gérées par le script). Après téléchargement, des versions \emph{écrêtées} (option \texttt{-{}-max-rows}) sont écrites dans \path{data/raw/cordis/} sous les noms canoniques suivants lorsque les jeux sont complets~:
\begin{itemize}[nosep]
  \item \path{h2020_projects_trimmed.csv}, \path{he_projects_trimmed.csv} (métadonnées projets)~;
  \item \path{h2020_organizations_trimmed.csv}, \path{he_organizations_trimmed.csv} (participations financées par projet).
\end{itemize}
Le parseur utilise le séparateur \textbf{point-virgule} et l'encodage \texttt{UTF-8-sig}, ce qui reflète la distribution standard CORDIS. Sur un dépôt partiel (par exemple uniquement Horizon~2020), seuls les fichiers effectivement téléchargés sont pris en compte dans l'étape de normalisation.

\subsubsection{OpenAlex}
Le script \path{src/fetch_openalex.py} interroge l'API REST \path{works} (\url{https://api.openalex.org/works}) avec un filtre sur le bailleur \textbf{Commission européenne} (\texttt{awards.funder\_id:F4320332161}), des paramètres \texttt{-{}-max-pages} et \texttt{-{}-per-page}, et une pagination par \texttt{cursor}. Le corpus collecté est stocké comme une \textbf{liste JSON} dans \path{data/raw/openalex/works_ec_funded.json}.

\subsubsection{OpenAIRE (optionnel)}
La collecte OpenAIRE est déclenchée uniquement lorsque \path{src/run_pipeline.py} est invoqué avec \texttt{-{}-include-openaire} (\path{src/fetch_openaire.py}) ; elle peut enrichir ou croiser avec d'autres identifiants, mais n'est pas requise pour le flux documenté ci-dessous.

\subsection{Traitement relationnel (\texorpdfstring{\texttt{clean\_normalize.py}}{clean\_normalize.py}) : logiques d'extraction}

\paragraph{Lecture CORDIS.}
Les tableaux projets et organisations sont lus avec détection souple des noms de colonnes (\texttt{pick\_column})~: par exemple \texttt{id} / \texttt{projectID} pour l'identifiant projet, \texttt{ecContribution} / \texttt{netEcContribution} pour le montant, \texttt{geolocation} pour \texttt{lat,lon}, etc. Les budgets textuels sont convertis en flottants (virgule remplacée si besoin).

\paragraph{Identifiant stable d'organisation (\texttt{org\_id}).}
Si CORDIS fournit un identifiant non vide (PIC, numéro interne, TVA, etc.), il est normalisé et concaténé au code pays pour former \texttt{org\_id}. Sinon, un hachage \textbf{SHA-1} (\texttt{digest[:10]}) est calculé sur la chaîne concaténée nom\texttt{|}ville\texttt{|}type afin de stabiliser les fusions lorsque l'identifiant source manque (\path{make_org_id}).

\paragraph{Projection ``co-projets''.}
Pour chaque projet regroupant au moins deux organisations, toutes les paires distinctes sont générées ; le fichier \path{edges_org_org_explicit.csv} contient une ligne par paire avec le poids \texttt{weight\_common\_projects} = nombre de projets en commun. Le fichier \path{edges_org_project.csv} matérialise la participation pondérée (montant européen) d'une organisation à un projet (\texttt{source\_org\_id}, \texttt{target\_project\_id}, \texttt{weight\_eur}). Le tableau \path{organizations.csv} agrège par organisation~: budget total perçu, nombre de projets distincts, coordonnées (géocode CORDIS si présent).

\paragraph{Ouverture via OpenAlex.}
Les institutions citées dans OpenAlex sont normalisées de la même manière que \texttt{org\_name\_norm} côté CORDIS. La jointure textuelle \textbf{exacte sur la normalisation du nom} alimente \path{edges_org_publication.csv} puis, via les concepts attachés aux œuvres, \path{edges_org_concept.csv} où le poids d'une paire (org, concept) est la \textbf{somme des scores OpenAlex} des concepts au fil des publications liées à l'organisation.

\subsection{Exports graphes après normalisation}

\subsubsection{Couche collaboration explicite (\texorpdfstring{\texttt{build\_graph.py}}{build_graph.py})}
Le graphe multiplexe est exporté en \textbf{GEXF} sous \path{data/graphs/collab_explicit.gexf}. Il représente organisations, projets et publications ainsi que les arêtes de financement, de co-participation et les liens attribution--publication. La couche utilisée dans \path{src/algorithms.py} pour PageRank ou Louvain est le \textbf{sous-graphe des organisations} lorsque les nœuds portent \texttt{node\_type\,=\,organization}, avec pondération selon les poids agrégés d'arête.

\subsubsection{Couche thématique (\texorpdfstring{\texttt{build\_thematic\_layer.py}}{build_thematic_layer.py})}
À partir de \path{edges_org_concept.csv}, on construit une matrice creuse \textbf{organisation $\times$ concept} où chaque coefficient est la somme des scores de concepts observés pour l'organisation. Les lignes sont normalisées en norme euclidienne (éviter division par zéro), puis la \textbf{similarité cosinus} entre chaque paire d'organisations $(i,j)$ est le produit scalaire des vecteurs unitaires correspondants. Une arête du graphe thématique implicite $G_{\mathrm{th}}$ est créée si la similarité $s_{ij}$ vérifie $s_{ij} \geq \theta$, avec $\theta=\texttt{0{,}30}$ par défaut (\texttt{-{}-threshold}). Le fichier produit est \path{data/graphs/thematic_implicit.gexf}. Les couples $(i,j)$ thématiquement liés mais \textbf{sans} arête dans \path{edges_org_org_explicit.csv} constituent des candidats préliminaires d'«~opportunité~».

\subsection{Opportunités de collaboration : définition quantitative}
\label{sec:opportunites}
Le script \path{src/gap_analysis.py} matérialise l'écart entre \emph{similarité thématique} et \emph{collaboration CORDIS observée}.

\paragraph{Sélection des paires candidates.}
Pour chaque arête de $G_{\mathrm{th}}$ de poids $s=\texttt{weight}$ (la similarité cosinus)~:
\begin{enumerate}[nosep]
  \item si $s < \texttt{min\_score}$ (valeur par défaut \textbf{0{,}40}, option \texttt{-{}-min-score}), la paire est rejetée ;
  \item si la même paire d'organisations apparaît déjà comme collaboration explicite dans \path{edges_org_org_explicit.csv}, elle est \textbf{exclue} (ce n'est pas un ``trou'', mais une collaboration avérée).
\end{enumerate}

\paragraph{Priorisation multi-critères (\texttt{gap\_analysis.py}).}
Notons $\mathrm{broker\_bonus}=1$ lorsque au moins une organisation appartient à l'ensemble des \emph{relays} défini comme intersection d'un fort betweenness dans le graphe explicite et d'un classement parmi les 30\,\% meilleures valeurs positives de contrainte de Burt sur le graphe thématique (\texttt{burt\_constraint\_thematic} issue de \path{organization_metrics.json}). Soient $c$ valant 1 si les deux pays diffèrent, $t$ valant 1 si les deux types juridiques diffèrent, et $\mathrm{balance}=\min\{n_{\mathrm{proj}}^{(a)},n_{\mathrm{proj}}^{(b)}\}\,/\,\max\{n_{\mathrm{proj}}^{(a)},n_{\mathrm{proj}}^{(b)}\}$ le rapport équilibrant les volumes projet. La priorité retenue est
\begin{equation}
\label{eq:prio-gap}
\pi \;=\; 0{,}60\, s \;+\; 0{,}15\, \mathrm{broker\_bonus} \;+\; 0{,}10\, c \;+\; 0{,}10\, t \;+\; 0{,}05\, \mathrm{balance}\,,
\end{equation}
puis coupée dans $[0,1]$ et arrondie. Les seuils métier suivants attribuent une étiquette \texttt{priority\_label}~: $\pi\geq 0{,}80$ «~Très forte~», $\geq 0{,}65$ «~Forte~», $\geq 0{,}55$ «~Intermédiaire~», sinon «~À surveiller~».

\paragraph{Livrables JSON.}
Jusqu'à \textbf{3\,000} fiches sont triées selon (\texttt{priority\_score} décroissant, \texttt{thematic\_score} décroissant) puis écrites dans \path{data/graphs/gap_analysis_top.json} (champs~: identifiants, scores, budgets, attributs booleans transfrontaliers/trans-types, rattache broker). Sur l'exemplaire mesuré pour ce rapport, le fichier contenait \textbf{1\,674} paires après filtrage—aussi moins que le plafonnement 3000 lorsque peu de couples passent successivement les seuils $\theta$ et \texttt{min\_score}.

\section{Difficultés rencontrées}

\begin{itemize}
  \item \textbf{Schémas CORDIS hétérogènes}~: mapping souple de colonnes et IDs manquants nécessitant un hachage composite pour \texttt{org\_id}.
  \item \textbf{Échelle}~: betweenness exacte irréaliste~; échantillonnage et bornes \texttt{-{}-max-edges} pour la carte.
  \item \textbf{Alignement OpenAlex $\leftrightarrow$ organisations}~: appariement par nom normalisé, faux positifs possibles.
  \item \textbf{Navigateur}~: HTML volumineux~; filtrage top-$N$ pour rester fluide.
  \item \textbf{\LaTeX{} XeLaTeX}~: résolution des polices TeX Gyre selon l'OS.
  \item \textbf{Dépendances Python}~: Torch / versions~; recours à \texttt{requirements-core.txt} sur certaines plateformes.
\end{itemize}

\begin{alertbox}
Les limites de \textbf{couverture} CORDIS/OpenAlex et d'interprétation causale des centralités demeurent~: voir bilan.
\end{alertbox}

\section{Bilan}

\subsection{Conclusion}
Ce projet formalise une chaîne reproductible \techterm{open data $\to$ graphe $\to$ métriques $\to$ visualisation} sur le financement européen, avec multiplexe explicite/thématique et interface cartographique. Les points forts incluent la modularité des scripts, une échelle réaliste (dizaines de milliers d'organisations, million d'arêtes collaboratives) et la double lecture contractuelle \emph{vs} similarité. La valeur combine \textbf{pipeline traçable} et \textbf{UX d'exploration}~; les métriques doivent être interprétées avec prudence (biais de données, pas de causalité directe).

\subsection{Perspectives}
Multiplex temporel, embeddings neuronaux optionnels (dépendances déjà listées), RAG sur fiches projet, évaluation utilisateur, API en lecture seule sur les JSON, alignement Leiden/cdlib pour comparaison des partitions.

\section{Cahier des charges}

\subsection{Contexte et périmètre}
Mettre en œuvre une chaîne \textbf{open data} reliant financements européens (CORDIS), publications et concepts (OpenAlex, en partie optionnel OpenAIRE), pour produire des \textbf{graphes analysables} (collaboration explicite, proximité thématique, écarts / opportunités) et des \textbf{visualisations navigables}.

\subsection{Exigences fonctionnelles}
\begin{itemize}
  \item \textbf{Ingestion}~: téléchargement ou dépôt contrôlé des jeux bruts, paramètres de volumétrie (\texttt{-{}-max-rows}, \texttt{-{}-max-pages}).
  \item \textbf{Normalisation}~: huit tables CSV UTF-8 sous \texttt{data/processed/} avec identifiants organisations stables.
  \item \textbf{Graphes}~: export GEXF multiplexe explicite, couche thématique implicite, JSON d'analyse d'écarts ; métriques (centralités, communautés, contrainte).
  \item \textbf{Livrables utilisateur}~: carte Folium HTML, fiches organisations, synthèses JSON (\texttt{metrics\_summary.json}, provenance), et page \texttt{interactive\_suite.html} pour accéder aux principales vues dans un seul navigateur.
\end{itemize}

\subsection{Exigences non fonctionnelles}
\begin{itemize}
  \item \textbf{Reproductibilité}~: scripts Python versionnés, environnement \texttt{venv} et \texttt{requirements.txt}.
  \item \textbf{Traçabilité}~: rapport de provenance (\texttt{data\_provenance\_report.json}) après vérification.
  \item \textbf{Performance praticable}~: limitation d'affichage navigateur tout en conservant les données complètes pour les analyses batch.
\end{itemize}

\subsection{Critères de réception (indicatifs)}
\begin{itemize}
  \item Pipeline \texttt{run\_pipeline.py} exécutable de bout en bout sur un échantillon documenté.
  \item Fichiers attendus présents~: CSV normalisés, \texttt{collab\_explicit.gexf}, \texttt{thematic\_implicit.gexf}, métriques, carte HTML, suite interactive.
\end{itemize}

\section{Manuel utilisateur}

\subsection{Objectif}
Permettre à un utilisateur de \textbf{regénérer les données}, \textbf{ouvrir la carte ou la suite HTML} et \textbf{interpréter} les couches sans relire l'intégralité du code.

\subsection{Lancer le pipeline}
\begin{itemize}
  \item Racine du dépôt, environnement activé.
  \item Commande recommandée~: \texttt{python src/run\_pipeline.py} avec bornes adaptées à la machine.
  \item En cas d'échec sur une étape, enchaîner manuellement les scripts du README~: \texttt{fetch\_*}, \texttt{clean\_normalize.py}, construction des graphes, \texttt{algorithms.py}, visualisation.
\end{itemize}

\subsection{Carte interactive et suite multi-vues}
\begin{itemize}
  \item Carte institutionnelle~: ouvrir \path{data/graphs/research_network_map_folium.html}.
  \item Vue unique (recommandée pour présentation)~: ouvrir \path{data/graphs/interactive_suite.html}, puis sélectionner l'onglet souhaité (collaborations pays, réseau européen, démo bipartite multi-projets).
  \item Recherche, zoom, bascule des couches~; depuis une popup organisation, suivre le lien vers \path{org_profiles/} si généré.
\end{itemize}

\subsection{Paramètres utiles}
\begin{itemize}
  \item \texttt{build\_thematic\_layer.py -{}-threshold}~: seuil de similarité cosinus sur le profil concepts (défaut~0,30).
  \item \texttt{gap\_analysis.py -{}-min-score}~: filtre minimal sur le graphe thématique pour les opportunités.
  \item \texttt{visualize\_folium.py -{}-max-edges}~: borne d'arêtes pour garder une carte fluide.
\end{itemize}

\subsection{Dépannage (aperçu)}
\begin{itemize}

  \item \textbf{iframes vides dans la suite}~: ouvrir \texttt{interactive\_suite.html} depuis le dossier \texttt{data/graphs/} (chemins relatifs vers les trois HTML).
\end{itemize}
